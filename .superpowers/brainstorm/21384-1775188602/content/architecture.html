SQLite SCHEMA
screenshots
├── id          INTEGER PRIMARY KEY
├── file_path   TEXT NOT NULL        -- original PNG location
├── ocr_text    TEXT                 -- extracted text (FTS5 indexed)
├── tag         TEXT DEFAULT 'other' -- code|conversation|article|notes|receipt|design|data|other
├── app_name    TEXT                 -- frontmost app at capture time
├── created_at  REAL NOT NULL        -- file creation timestamp
└── thumbnail   BLOB                 -- small JPEG thumbnail for grid view

screenshots_fts (FTS5 virtual table on ocr_text)
